Bisection Method in C

#include 

float f(float x) {
  return x*x*x - x - 2;
}

int main() {
  float a = 1, b = 2, c;
  while ((b-a) >= 0.01) {
    c = (a+b)/2;
    if (f(c) == 0.0)
      break;
    else if (f(c)*f(a) < 0)
      b = c;
    else
      a = c;
  }
  printf("Root = %.4f\n", c);
  return 0;
}
